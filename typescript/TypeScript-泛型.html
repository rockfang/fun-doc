# 普通用法

function mergeObject<A, B>(obj1: A, obj2: B) {
  return Object.assign(obj1, obj2)
}

const mergeResult = mergeObject({ name: 'cai' }, { company: 'hcdsj' })
console.log(mergeResult.company)

# 泛型限制

尽管泛型指定通用的类型，但我们可以给通用的类型做限制，提升代码健壮性。

function mergeObject<A extends object, B extends object>(obj1: A, obj2: B) {
  return Object.assign(obj1, obj2)
}

const mergeResult = mergeObject({ name: 'cai' }, { company: 'hcdsj' })
// 使用数字等其他类型变量时，会提示错误
// const mergeResult2 = mergeObject({ name: 'cai' }, 123)
console.log(mergeResult.company)

# 泛型限制-自定义类型

定义一个方法返回拥有length属性的参数的描述信息和参数本身。如下：

// type lengthy = {
// 	length: number
// }
interface lengthy {
	length: number
}
function countAndDescription<T extends lengthy>(e: T):[T,string] {
	let descriptions = ''
	if (e.length === 1) {
		descriptions = 'the length is one'
	} else {
		descriptions = `have ${e.length} elements`
	}
	return [e, descriptions]
}
const str = 'This is a sentence'
console.log(countAndDescription(str))//["This is a sentence", "have 18 elements"]

const arr = [3,1,2]
console.log(countAndDescription(arr));//[ [3, 1, 2], "have 3 elements"]

console.log(countAndDescription(1));// Error

# 泛型限制-keyof 类型

keyof 获取某种类型的所有键，其返回类型是联合类型

type t = keyof { a: 3; b: 2 } // 同 type t = "a" | "b"

function extractAndConvert<T extends object, K extends keyof T>(
	obj: T,
	key: K
) {
	return 'value:' + obj[key]
}

const labelInfo = {

    labelName: '活跃度'

# 泛型类

封装一个数组存取的类：

class CustomStorage<T extends string | number | boolean> {
	private data: T[] = []

	add(item: T) {
		this.data.push(item)
	}

	remove(item: T) {
		this.data.splice(this.data.indexOf(item), 1)
	}

	getAll() {
		return [...this.data]
	}
}
const store = new CustomStorage<string>()
store.add('token')
store.add('account')
console.log(store.getAll())
store.remove('account')
console.log(store.getAll());

示例2：使用Partial实现对象属性覆盖

interface Todo {
  title: string;
  description: string;
  done: boolean;
}

function updateTodo(todo: Todo, newTodo: Partial<Todo>) {
  return { ...todo, ...newTodo };
}

const todo: Todo = {
  title: 'First Todo',
  description: 'this is the first todo',
  done: false
};

updateTodo(todo, { done: true });

# 总结

泛型及泛型限制使得我们可以灵活的组合各种类型，实现类型安全使用

泛型和联合类型的场景：

联合类型：类中每个方法都想灵活的使用任意类型

泛型：整个类中锁定一个确定类型使用
