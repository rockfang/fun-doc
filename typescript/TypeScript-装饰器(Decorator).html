利用装饰器做页面渲染：

function withTemplate(template: string, container: string) {
	return function (constructor: any) {
        const parent = document.querySelector(container)
        const p = new constructor("default")
		if (parent) {
            parent.innerHTML = template
            parent.querySelector('h1')!.textContent = p.name
		}
	}
}

# 装饰器中返回Class

处理在类实例化时才调用装饰器的方法，可以通过在装饰器中返回class的方式实现：

function withTemplate(template: string, container: string) {
	return function <T extends { new (...args: any[]): { name: string } }>(
		constructor: T
	) {
		return class extends constructor {
			constructor(...args: any[]) {
				super(args)
				const parent = document.querySelector(container)
				if (parent) {
					parent.innerHTML = template
					parent.querySelector('h1')!.textContent = this.name
				}
			}
		}
	}
}
@withTemplate("<h1 style='color:red'>hello</h1>", '#app')
class Person {
	name: string = 'default'
	constructor(name: string) {
		this.name = name
	}
}
const p = new Person('fun')

# 使用decorator实现方法bind

class Printer {
	message = 'Go Go Go'
	showMessage() {
		console.log(this.message)
	}
}

const printer = new Printer()
const clickButton = document.querySelector('button')
clickButton?.addEventListener('click', printer.showMessage) //undefined 因为this指向事件本身了
clickButton?.addEventListener('click', printer.showMessage.bind(printer)) //GO GO GO

上面代码，由于在addEventListener回调方法中调用this，this指向变化，导致直接调用方法达不到预期效果。常规方式需手动 method.bind(obj)方式达到预期效果。

采用方法装饰器实现该功能：

//function decorator参数形式
function autoBind(
	target: any,
	methodName: string,
	descriptor: PropertyDescriptor
) {
	const originalMethod = descriptor.value
	const adjDescriptor: PropertyDescriptor = {
		configurable: true,
		enumerable: false,
		get() {
			const boundFn = originalMethod.bind(this) //this只想get方法的调用者(对象.属性),即对象本身
			return boundFn
		},
	}
	return adjDescriptor
}
class Printer {
	message = 'Go Go Go'

	@autoBind
	showMessage() {
		console.log(this.message)
	}
}

const printer = new Printer()
const clickButton = document.querySelector('button')
clickButton?.addEventListener('click', printer.showMessage) // GO GO GO

写装饰器之前，可以先看下 TypeScript索引签名使用open in new window

//构建这样一种数据结构描述validate信息
//结构为： constructor.name : {propName: ['required','positive',...]}
interface ValidatorConfig {
	[property: string]: {
		[validatableProp: string]: string[] //['required','positive']
	}
}

const registeredValidators: ValidatorConfig = {}

function required(target: any, propName: string) {
	registeredValidators[target.constructor.name] = {
		...registeredValidators[target.constructor.name],
		[propName]: ['required'],
    }
}

function positiveNumber(target: any, propName: string) {
	registeredValidators[target.constructor.name] = {
		...registeredValidators[target.constructor.name],
		[propName]: ['positive'],
    }
    console.log('positiveNumber ' + propName,registeredValidators);
}
//校验指定对象的属性
function validate(obj: any) {
	// 拿到该类(通过构造方法名)的 校验信息
    const objValidatorConfig = registeredValidators[obj.constructor.name]
	if (!objValidatorConfig) {
		return true
	}

	let isValid = true
	for (const prop in objValidatorConfig) {
		for (const validator of objValidatorConfig[prop]) {
			switch (validator) {
				case 'required':
					isValid = isValid && !!obj[prop] //判断该属性是否存在
					break
				case 'positive':
					isValid = isValid && obj[prop] > 0
			}
		}
	}
	return isValid
}

class Course {

	@required
    title: string
    
	@positiveNumber
	price: number

	constructor(t: string, p: number) {
		this.title = t
		this.price = p
	}
}

document.querySelector('#submit-course')?.addEventListener('click', () => {
	const titleDom = document.querySelector('#title') as HTMLInputElement
	const priceDom = document.querySelector('#price') as HTMLInputElement

    const course = new Course(titleDom.value, +priceDom.value)
    console.log(validate(course));
    
	if (!validate(course)) {
		alert('Not valid!!')
		return
	}
	console.log(course)
})
