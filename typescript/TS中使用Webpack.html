    "build": "webpack",
    "start": "lite-server"

    "lite-server": "^2.5.4",

    "typescript": "^4.1.3",
    "webpack": "^4.41.2",

# 新建webpack.config.js并配置如下：

const path = require("path");
module.exports = {
  entry: "./src/app.ts",
  output: {
    filename: "bundle.js",
    path: path.resolve(__dirname, "dist"),
  },
  devtool: 'inline-source-map',
  module: {
    rules: [
      {
        test: /\.ts$/,
        use: "ts-loader",
        exclude: /node_modules/,
      },
    ],
  },
  resolve: {
    extensions: [".ts", ".js"],
  },
};

# 创建生产环境的webpack配置

# 创建生产环境配置文件webpack.config.prod.js

const path = require("path");
const CleanPlugin = require("clean-webpack-plugin")
module.exports = {
  entry: "./src/app.ts",
  output: {
    filename: "bundle.js",
    path: path.resolve(__dirname, "dist"),
  },
  devtool: 'none',
  module: {
    rules: [
      {
        test: /\.ts$/,
        use: "ts-loader",
        exclude: /node_modules/,
      },
    ],
  },
  resolve: {
    extensions: [".ts", ".js"],
  },

  plugins: [
        new CleanPlugin.CleanWebpackPlugin()
  ]
};
