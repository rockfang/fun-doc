abstract class TreeInfo {
	static DOMAIN_SYS_ID = 1
	constructor(protected readonly domainId: number, public treeJson: string) {}
	abstract logTreeInfo(): void

	static createInfoObject(id: string | number, info: string) {
		return { key: id, value: info }
	}
}

class SubTree extends TreeInfo {

	public set mPid(v: number) {
		this.parentId = v
    }
    
    logTreeInfo(): void {

单例模式：

class SubTree extends TreeInfo {
    private parentId: number
    private static instance: SubTree
	private constructor(domainId: number, treeJson: string, parentId: number) {
		super(domainId, treeJson)
		this.parentId = parentId
    }
    static getInstance() {
        if(!SubTree.instance) {
            this.instance = new SubTree(5,'singleton treejson',1)
        }
        return this.instance
    }
	get idInfo() {
		return this.domainId + '---' + this.parentId
	}

	public set mPid(v: number) {
		this.parentId = v
    }
    
    logTreeInfo(): void {
        console.log(`subTree parentId ${this.parentId}`);
    }
}
const st = SubTree.getInstance()
st.logTreeInfo()

总结：

关键字用途
private私有，不能再对象外赋值属性
public公有属性
protected受保护属性，允许子类访问
readonly只读，不能改变该属性值。
set定义set方法，通过属性值方式赋值
get定义get方法，通过属性值方式访问
extends继承
static静态变量/方法，通过类名访问
abstract定义抽象类，抽象方法
单例模式使用私有构造方法和静态方法实现

接口仅用于描述对象的结构，非常清晰

可由子类实现,实现多个接口

接口之间可以继承，继承多个接口

相比抽象类，接口中不能包含任何非抽象方法

enum LogType {
	SUCCESS,
	WARNING,
	INFO,
	ERROR,
}
interface CommonLog {
	type: LogType
	message: string
	title: string

    log(content: string): void
}

class ColorLog implements CommonLog {
	type: LogType
	message: string
	title: string
	constructor(type: LogType, title: string, message: string) {
		this.type = type
		this.message = message
		this.title = title
	}
	log(content: string): void {
		if (this.type === LogType.SUCCESS) {
			console.log(`%c ${content}`, 'color:#0f0;')
		} else if (this.type === LogType.ERROR) {
			console.log(`%c ${content}`, 'color:red;')
		} else if (this.type === LogType.WARNING) {
			console.log(`%c ${content}`, 'color:orange;')
		} else {
			console.log(content)
		}
	}
}
let colorLog = new ColorLog(LogType.SUCCESS, 'warning title', 'waring content')
colorLog.log("waring content")

接口中的可选参数

类和接口都可以使用可选参数/方法

interface Uname {
    name?: string,
    say():void
}
class subName implements Uname {
	firstName: string
	lastName?: string
	constructor(firstName: string, lastName?: string) {
		this.firstName = firstName
		this.lastName = lastName
	}
    say(): void {
        console.log(this.firstName)
    }
}
