# 交叉类型

交叉类型（Intersection Types）是将多个类型合并为一个类型，包含了所需的所有类型的特性

type Admin = {
	name: string
	privileges: string[]
}

type Employee = {
	name: string
	registTime: Date
}

type Manager = Admin & Employee

const m: Manager = {
	name: 'fun',
	privileges: ['manager'],
	registTime: new Date(),
}

type Combinable = string | number
type Numeric = number | boolean

type Universal = Combinable & Numeric
const u: Universal = 9527

interface Bird {
	type: 'bird'

	type: 'horse'
	groundSpeed: number

type Animal = Bird | Horse

function moveAimal(animal: Animal) {
	let speed
	switch (animal.type) {
		case 'bird':
            speed = animal.flyingSpeed
			break
		case 'horse':
            speed = animal.groundSpeed
			break
    }
    console.log(`Moving speed is: ${speed}`)
}
moveAimal({'type':'bird',flyingSpeed:213})

# 函数重载

先看这样的场景，定义了一个addFn函数

type Combinable = string | number
const addFn = (a: Combinable, b: Combinable) => {
	if (typeof a === 'string' || typeof b === 'string') {
		return a.toString() + b.toString()
	}
	return a + b
}
const myName = addFn("cai"," fun")
myName.split(" ")

function addFn(a: string, b: string): string
function addFn(a: number, b: number): number
function addFn(a: number, b: string): string
function addFn(a: string, b: number): string
function addFn(a: Combinable, b: Combinable) {
	if (typeof a === 'string' || typeof b === 'string') {
		return a.toString() + b.toString()
	}
	return a + b
}
const myName = addFn('cai', ' fun')
myName.split(' ')

# 空值合并

Nullish Coalescing

使用??符，实现null和undefined的合并

const result = {type: undefined}

const msg = result.type ?? 2//若type为0或者'',都会认为是有效值
console.log(msg);

const msg2 = result.type && 2//若type为0或者'',会取后面的值
console.log(msg);
console.log(msg2);
