# 事件添加和使用

添加事件，和动态数据使用:

  • 采用class方式定义组件
  • 使用{}传递参数
  • 只能通过setState改变state值 (类似vuex中state管理方式)
class App extends Component {
  state = {
    persons: [
      {
        name: "Alan",
        age: 23,
      },
      {
        name: "Ackerman",
        age: 23,
      },
      {
        name: "ming",
        age: 18,
      },
    ],
  };

  changeList = () => {
    this.setState({
      persons: [
        {
          name: "Alan boy",
          age: 16,
        },
        {
          name: "Ackerman",
          age: 16,
        },
        {
          name: "ming",
          age: 16,
        },
      ],
    });
  };
  render() {
    return (
      <div className="App">
        <header className="App-header">
          <img src={logo} className="App-logo" alt="logo" />
          <h2>This is React</h2>
          <button onClick={this.changeList}>切换</button>
          <Person
            name={this.state.persons[0].name}
            age={this.state.persons[0].age}
          >
            yes
          </Person>
          <Person
            name={this.state.persons[1].name}
            age={this.state.persons[1].age}
          >
            yes
          </Person>
          <Person
            name={this.state.persons[2].name}
            age={this.state.persons[2].age}
          >
            yes
          </Person>
        </header>
      </div>
    );
  }
}

# 函数式组件中使用state

使用useState 将获得一个数组，得到当前state和改变state的方法；

import { useState } from 'react'
const Person = (props) => {
  const [personState, setPersonState] = useState({
    gender: 'male',
    hobbies: ['baseball'],
  })
  const switchGender = () => {
    setPersonState({
      gender: 'female',
    })
  }

  return (
    <div>
      <div>hello {props.name}</div>
      <div>{props.children}</div>
      <br />
      <div>{personState.gender}</div>
      <div>{personState.hobbies}</div>
      <button onClick={switchGender}>改变性别</button>
    </div>
  )
}
export default Person

# radium使用

1

Radium is a set of tools to manage inline styles on React elements. It gives you powerful styling capabilities without CSS.

import Radium from 'radium';
import React from 'react';
import color from 'color';
 
class Button extends React.Component {
  static propTypes = {
    kind: PropTypes.oneOf(['primary', 'warning']).isRequired
  };
 
  render() {
    // Radium extends the style attribute to accept an array. It will merge
    // the styles in order. We use this feature here to apply the primary
    // or warning styles depending on the value of the `kind` prop. Since its
    // all just JavaScript, you can use whatever logic you want to decide which
    // styles are applied (props, state, context, etc).
    return (
      <button style={[styles.base, styles[this.props.kind]]}>
        {this.props.children}
      </button>
    );
  }
}
 
Button = Radium(Button);
 
// You can create your style objects dynamically or share them for
// every instance of the component.
var styles = {
  base: {
    color: '#fff',
 
    // Adding interactive state couldn't be easier! Add a special key to your
    // style object (:hover, :focus, :active, or @media) with the additional rules.
    ':hover': {
      background: color('#0074d9')
        .lighten(0.2)
        .hexString()
    }
  },
 
  primary: {
    background: '#0074D9'
  },
 
  warning: {
    background: '#FF4136'
  }
};

import classes from "./Person.css";

  render() {
    console.log("[Person.js] rendering...");

          <p>{this.props.children}</p>
          <input
            type="text"
            onChange={this.props.changed} 
            value={this.props.name}
          />
        </div>
      </React.Fragment>
    );
  }
}

export default Person;
